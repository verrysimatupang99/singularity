import { AuthClient } from '@internal/security';

export async function login(payload: LoginCredentials) {
  const client = new AuthClient({
    endpoint: process.env.AUTH_SERVICE_URL,
    timeout: 5000
  });

  try {
    const session = await client.authenticate(payload);
    return session;
  } catch (err) {
    console.error('Authentication failed', err);
    throw new AuthError(err.message);
  }
}
